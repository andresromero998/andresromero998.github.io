df_sim$pscore_hat <- predict(ps_mod, type = "response")

  1 / df_sim$pscore_hat,
  1 / (1 - df_sim$pscore_hat))

  sum(D*Y - (1-D)*Y*(pscore_hat/(1-pscore_hat))) / sum(D)

# ── Plug-in (8 cells) ─────────────────────────────────────────────────────────
# D = treated group, t = post, S = subgroup (e.g. affected = 1)
cells <- df |>
  group_by(D, t, S) |>
  summarise(y = mean(Y), .groups = "drop")

get_cell <- function(d, tt, s)
  cells$y[cells$D==d & cells$t==tt & cells$S==s]

DID_S1 <- (get_cell(1,1,1) - get_cell(1,0,1)) -
           (get_cell(0,1,1) - get_cell(0,0,1))

Sant'Anna & Zhao (2020) — Doubly-Robust Triple DiD drdid · manual
Extends the Sant'Anna & Zhao DR-DiD estimator to a DDD setting by stratifying on the third dimension S. The DDD effect is recovered as ATT(S=1) − ATT(S=0), each estimated with doubly-robust IPW+OR weighting. Villavicencio (2023) uses this logic: estimate separate DR-DiD for the subgroup of interest and a placebo subgroup, then difference them.
# ── Step 1: Estimate DR-DiD separately for S = 1 and S = 0 ────────────────── # Keep only pre/post periods; panel data (same units observed twice) df_wide <- df |> select(unit_id, year, Y, treated, S, X1, X2) |> pivot_wider(names_from = year, values_from = Y, names_prefix = "Y_") # → Y_pre, Y_post columns # Subgroup S = 1 (e.g. "affected" units) df_s1 <- df_wide |> filter(S == 1) dr_s1 <- drdid_panel( y1 = df_s1$Y_post, y0 = df_s1$Y_pre, D = df_s1$treated, covariates = as.matrix(df_s1[, c("X1", "X2")]), inffunc = TRUE # keep influence functions for inference ) # Subgroup S = 0 (placebo / "unaffected" units) df_s0 <- df_wide |> filter(S == 0) dr_s0 <- drdid_panel( y1 = df_s0$Y_post, y0 = df_s0$Y_pre, D = df_s0$treated, covariates = as.matrix(df_s0[, c("X1", "X2")]), inffunc = TRUE ) # ── Step 2: DDD = ATT(S=1) − ATT(S=0) ─────────────────────────────────────── DDD_dr <- dr_s1$ATT - dr_s0$ATT # Delta-method SE using influence functions # (assumes balanced panels; rescale by n_s / n if unbalanced) inf_ddd <- dr_s1$inf.func - dr_s0$inf.func[seq_len(nrow(df_s1))] se_ddd <- sqrt(mean(inf_ddd^2) / nrow(df_wide)) ci_lo <- DDD_dr - 1.96 * se_ddd ci_hi <- DDD_dr + 1.96 * se_ddd cat(sprintf("DR-DDD: %.4f SE: %.4f 95%%CI: [%.4f, %.4f]\n", DDD_dr, se_ddd, ci_lo, ci_hi)) # ── Step 3 (staggered): apply logic per cohort g using did + drdid ──────────── # For each cohort g, split df into S=1 and S=0 and run att_gt(), # then difference the aggregated dynamic effects. library(did) run_cs_subgroup <- function(data, s_val) { att_gt( yname = "Y", tname = "year", idname = "unit_id", gname = "first_treat", control_group = "nevertreated", est_method = "dr", # Sant'Anna & Zhao inside each subgroup base_period = "universal", data = filter(data, S == s_val) ) } cs_s1 <- run_cs_subgroup(df, 1) cs_s0 <- run_cs_subgroup(df, 0) # Dynamic aggregation per subgroup dyn_s1 <- aggte(cs_s1, type = "dynamic") dyn_s0 <- aggte(cs_s0, type = "dynamic") # Manual DDD event-study: diff the period-specific ATTs ddd_dynamic <- tibble::tibble( e = dyn_s1$egt, att_s1 = dyn_s1$att.egt, att_s0 = dyn_s0$att.egt[match(dyn_s1$egt, dyn_s0$egt)], ddd = att_s1 - att_s0, se_ddd = sqrt(dyn_s1$se.egt^2 + dyn_s0$se.egt^2) ) |> mutate(lo = ddd - 1.96*se_ddd, hi = ddd + 1.96*se_ddd) # Plot DR-DDD event study ggplot(ddd_dynamic, aes(e, ddd)) + geom_ribbon(aes(ymin = lo, ymax = hi), fill = "#f472b6", alpha = .2) + geom_line(colour = "#f472b6") + geom_point(colour = "#f472b6") + geom_hline(yintercept = 0, linetype = "dashed", colour = "grey60") + geom_vline(xintercept = -.5, linetype = "dotted") + theme_minimal() + labs(x = "Periods relative to treatment", y = "DR-DDD estimate (95% CI)", title = "Sant'Anna & Zhao DR — Triple DiD Event Study", subtitle = "ATT(g,t | S=1) − ATT(g,t | S=0), doubly robust")
Ortiz-Villavicencio & Sant'Anna (2025) — triplediff package triplediff
Native DDD estimators with RA, IPW, and doubly-robust identification under a conditional DDD parallel-trends assumption. Key argument: pname = the eligibility/partition variable (the third dimension, e.g. gender, sector, age group). Handles two-period and staggered multi-period designs; UX mirrors the did package.
Paper: Ortiz-Villavicencio & Sant'Anna (2025) — arXiv:2505.09942. Why it matters: standard DDD-of-DiDs and 3-way FE regressions are generally invalid when covariates are needed for identification; this package fixes that.

  gname         = "state",          # first treatment period (0 = never)

  xformla       = ~ cov1 + cov2 + cov3 + cov4,

library(stm)
stm_in <- convert(dfm_obj, to = "stm")

# Fit STM with prevalence covariate
stm_mod <- stm(
  documents  = stm_in$documents,
  vocab      = stm_in$vocab, K = 50,
  prevalence = ~ party + s(day_of_year),
  data       = stm_in$meta,
  init.type  = "Spectral", verbose = FALSE
)
labelTopics(stm_mod, n = 10)

# Topic-covariate effects
eff <- estimateEffect(1:50 ~ party + s(day_of_year),
                      stm_mod, meta = stm_in$meta)
plot(eff, covariate = "party", topics = c(1,5,10),
     model = stm_mod, method = "difference",
     cov.value1 = "Republican", cov.value2 = "Democrat")
